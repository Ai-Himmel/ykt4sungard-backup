71          BufferedImage img = null;
72          try {
73              img = ImageIO.read(ClassLoader.getSystemResource(imgPath));
74          } catch (IOException e) {
75              logger.error("找不到图片: " + imgPath);

92      public static TexturePaint createTexturePaint(String imgPath) {
93          ImageIcon icon = getImageIcon(imgPath);
94          int imageWidth = icon.getIconWidth();
95          int imageHeight = icon.getIconHeight();
96          BufferedImage bi = new BufferedImage(imageWidth, imageHeight, 2);
97          Graphics2D g2d = bi.createGraphics();
98          g2d.drawImage(icon.getImage(), 0, 0, null);
99          g2d.dispose();
100         return new TexturePaint(bi, new Rectangle(0, 0, imageWidth, imageHeight));
101     }
