148      * 签  名：int __stdcall ks_card(char *sCardPhyID);
149      * 功  能：寻卡
150      * 参  数: char *sCardPhyID 输出参数，返回卡的物理序列号,长度17字节。
151      * 返回值: 返回0为成功，其他为错误
152      *
153      * @param result    .
154      * @param showError 是否记录寻卡失败日志，showError=false时适用于某些场景，比如连续不断的自动寻卡，此时寻卡失败是由于用户未放置卡，无须连续的显示错误信息
155      * @return .
156      */
157     public static boolean ksCard(Properties result, boolean showError) {
158         boolean success = false;
159         Pointer pCardPhyId = null;
160         try {
161             ksCard = new JNative(KS_CARD_DLL, CARD);
162             ksCard.setRetVal(Type.INT);
163             pCardPhyId = new Pointer(MemoryBlockFactory.createMemoryBlock(100));
164             int i = 0;
165             ksCard.setParameter(i++, pCardPhyId);
166 
167             logger.debug("invoke method " + CARD);
168             ksCard.invoke();
169             int returnValue = ksCard.getRetValAsInt();
170             logger.debug(CARD + "[returnValue=" + returnValue + "]");
171 
172             if (returnValue == 0) {
173                 success = true;
174                 String cardPhyId = pCardPhyId.getAsString();
175                 result.setProperty("cardPhyId", cardPhyId);
176             } else {
177                 if (showError) {
178                     logger.error(ksGetErrMsg());
179                 }
180             }
181         } catch (NativeException e) {
182             logger.error(e.getMessage());
183         } catch (IllegalAccessException e) {
184             logger.error(e.getMessage());
185         } finally {
186             dispose(pCardPhyId);
187         }
188         return success;
189     }
