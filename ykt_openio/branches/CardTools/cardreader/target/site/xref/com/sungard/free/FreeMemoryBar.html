15  public class FreeMemoryBar extends FreeProgressBar {

20      private static NumberFormat format = NumberFormat.getInstance();
21      private int delay = 2000;
22  
23      public FreeMemoryBar() {
24          super(0, 0, 100);
25          ActionListener taskPerformer = new ActionListener() {
26              public void actionPerformed(ActionEvent evt) {
27                  final long usedMemory = memorymbean.getHeapMemoryUsage().getUsed();
28                  final long totalMemory = memorymbean.getHeapMemoryUsage().getMax();
29                  updateMemoryUsage(usedMemory, totalMemory);
30              }
31          };
32          new Timer(delay, taskPerformer).start();
33      }
34  
35      private void updateMemoryUsage(long usedMemory, long totalMemory) {
36          int percent = (int) (usedMemory * 100 / totalMemory);
37          this.setValue(percent);
38          String usedMega = format.format(usedMemory / kilo / kilo) + mega;
39          String totalMega = format.format(totalMemory / kilo / kilo) + mega;
40          String message = usedMega + "/" + totalMega;
41          this.setString(message);
42  
43          this.setToolTipText("Memory used " + format.format(usedMemory) + " of total " + format.format(totalMemory));
44      }
