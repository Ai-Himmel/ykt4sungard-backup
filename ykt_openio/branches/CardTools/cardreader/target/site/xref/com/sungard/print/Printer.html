45      public boolean print() {
46          boolean success = false;
47          try {
48              //指定打印输出格式
49              DocFlavor flavor = DocFlavor.SERVICE_FORMATTED.PRINTABLE;
50              //设置打印属性
51              PrintRequestAttributeSet parameters = new HashPrintRequestAttributeSet();
52              // 设置纸张大小,也可以新建MediaSize类来自定义大小
53              //宽56mm 高100mm
54              int pageWidth = Configuration.getIntProperty(Constant.PAGE_WIDTH, 50);
55              int pageHeight = Configuration.getIntProperty(Constant.PAGE_HEIGHT, 100);
56              MediaSize mediaSize = new MediaSize(pageWidth, pageHeight, Size2DSyntax.MM, MediaSizeName.INVOICE);
57              parameters.add(MediaSizeName.INVOICE);
58  //            parameters.add(MediaSizeName.ISO_A4);
59  
60              DocAttributeSet docAttributeSet = new HashDocAttributeSet();
61  
62              PrintService printService = getPrintService(flavor, parameters);
63              if (printService == null) {
64                  logger.error("Can't find printer");
65              } else {
66                  //指定打印内容
67                  Doc doc = new SimpleDoc(document, flavor, docAttributeSet);
68                  //创建打印作业
69                  DocPrintJob job = printService.createPrintJob();
70                  //进行每一页的具体打印操作
71                  job.print(doc, parameters);
72                  success = true;
73                  logger.info("Using printer[" + printService.getName() + "], start printing");
74              }
75          } catch (PrintException e) {
76              logger.error("打印失败:" + e.getMessage());
77          }
78          return success;
79      }

82       * 寻找打印机
83       *
84       * @param flavor     .
85       * @param parameters .
86       * @return
87       */
88      public PrintService getPrintService(DocFlavor flavor, PrintRequestAttributeSet parameters) {
89          PrintService printService;
90  
91          //获得默认打印机,如果用户在配置文件中配置了打印机名称,那么将配置的打印机作为默认打印机,否则寻找系统默认打印机
92          PrintService defaultPrintService;
93          String printServiceName = Configuration.getProperty(Constant.PRINT_SERVICE);
94          if (StringUtils.isNotBlank(printServiceName)) {
95              logger.info("start finding printer " + printServiceName);
96              HashAttributeSet attributeSet = new HashAttributeSet();
97              attributeSet.add(new PrinterName(printServiceName, null));
98              PrintService[] printServices = PrintServiceLookup.lookupPrintServices(null, attributeSet);
99              defaultPrintService = printServices != null && printServices.length > 0 ? printServices[0] : null;
100         } else {
101             defaultPrintService = PrintServiceLookup.lookupDefaultPrintService();
102         }
103 
104         //如果showDialog为true,那么弹出对话框由用户自己选择打印机
105         String showDialog = Configuration.getProperty(Constant.SHOW_DIALOG);
106         if ("true".equals(showDialog)) {
107             printService = ServiceUI.printDialog(null, 100, 200, PrintServiceLookup.lookupPrintServices(null, null), defaultPrintService, flavor, parameters);
108         } else {
109             printService = defaultPrintService;
110         }
111 
112         return printService;
113     }
