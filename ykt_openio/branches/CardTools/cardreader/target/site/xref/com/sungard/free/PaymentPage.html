215     public void sum() {
216         //自动计算合计金额
217         BigDecimal totalFee = BigDecimal.valueOf(0);
218         if (verify(medicineText, medicineTipLabel) && StringUtils.isNotBlank(medicineText.getText())) {
219             totalFee = totalFee.add(BigDecimal.valueOf(Double.parseDouble(medicineText.getText())));
220         }
221         if (verify(dressingText, dressingTipLabel) && StringUtils.isNotBlank(dressingText.getText())) {
222             totalFee = totalFee.add(BigDecimal.valueOf(Double.parseDouble(dressingText.getText())));
223         }
224         if (verify(injectionText, injectionTipLabel) && StringUtils.isNotBlank(injectionText.getText())) {
225             totalFee = totalFee.add(BigDecimal.valueOf(Double.parseDouble(injectionText.getText())));
226         }
227         if (verify(vaccineText, vaccineTipLabel) && StringUtils.isNotBlank(vaccineText.getText())) {
228             totalFee = totalFee.add(BigDecimal.valueOf(Double.parseDouble(vaccineText.getText())));
229         }
230         if (verify(otherText, otherTipLabel) && StringUtils.isNotBlank(otherText.getText())) {
231             totalFee = totalFee.add(BigDecimal.valueOf(Double.parseDouble(otherText.getText())));
232         }
233         totalText.setText(String.valueOf(totalFee));
234     }

236     public boolean verify(JTextField inputText, JLabel errorLabel) {
237         boolean valid = false;
238         try {
239             if (StringUtils.isNotBlank(inputText.getText())) {
240                 Object value = feeFormat.parseObject(inputText.getText());
241                 inputText.setText(value.toString());
242             }
243             valid = true;
244             errorLabel.setText("");
245         } catch (ParseException e) {
246             errorLabel.setForeground(Color.red);
247             errorLabel.setText("请输入合法数字");
248             inputText.requestFocusInWindow();
249         }
250         return valid;
251     }

304     private void chargeAction(ActionEvent e) {
305         String tramsAmt = totalText.getText();
306         if (StringUtils.isBlank(tramsAmt)) {
307             JOptionPane.showMessageDialog(this, "请输入扣款金额", "Warning", JOptionPane.WARNING_MESSAGE);
308         } else {
309             int response = JOptionPane.showConfirmDialog(this, "应支付金额【" + tramsAmt + "元】，是否扣款？", "扣款确认", JOptionPane.YES_NO_OPTION);
310             if (0 == response) {
311                 String cardPhyId = result.getProperty("cardPhyId");
312                 int cardno = Integer.parseInt(result.getProperty("CARDNO"));
313                 int cardBal = Integer.parseInt(result.getProperty("CARDBAL"));
314                 int payCnt = Integer.parseInt(result.getProperty("PAYCNT"));
315                 boolean success = false;
316                 if (KsCardInterfaceUtil.ksPayPrepare(cardPhyId, cardno, cardBal, payCnt, Double.valueOf(Double.valueOf(tramsAmt) * 100).intValue(), result)) {
317                     if (KsCardInterfaceUtil.ksCardBalanceDec(result)) {
318                         if (KsCardInterfaceUtil.ksPayProcess()) {
319                             success = true;
320                         }
321                     }
322                 }
323                 if (success) {
324                     String refNo = result.getProperty("refNo");
325                     String mac = result.getProperty("mac");
326                     refNoText.setText(refNo);
327                     macText.setText(mac);
328                     KsCardInterfaceUtil.getFieldValue("CARDBAL", result);
329                     balanceText.setText(String.valueOf(Double.parseDouble(result.getProperty("CARDBAL")) / 100));
330                     chargeBtn.setEnabled(false);
331                     reverseBtn.setEnabled(true);
332                     Payrecord payrecord = getPayrecord();
333                     //1 扣款，0 冲正
334                     payrecord.setTransType("1");
335                     payrecord.setClassName(PayrecordService.findClassName(payrecord.getStuempno()));
336                     if (PayrecordService.savePayrecord(payrecord)) {
337                         KsCardInterfaceUtil.ksBeep();
338                         //打印
339                         String printable = Configuration.getProperty(Constant.PRINTABLE);
340                         if ("true".equals(printable)) {
341                             print(payrecord);
342                         }
343                     } else {
344                         JOptionPane.showMessageDialog(this, "保存付费记录失败", "Error", JOptionPane.ERROR_MESSAGE);
345                     }
346                 } else {
347                     JOptionPane.showMessageDialog(this, "扣款失败", "Error", JOptionPane.ERROR_MESSAGE);
348                 }
349             }
350         }

404     private void reverseAction(ActionEvent e) {
405         int cardno = Integer.parseInt(result.getProperty("CARDNO"));
406         int cardBal = Integer.parseInt(result.getProperty("CARDBAL"));
407         KsCardInterfaceUtil.getFieldValue("DPSCNT", result);
408         int dpsCnt = Integer.parseInt(result.getProperty("DPSCNT"));
409         String refNo = refNoText.getText();
410         String mac = macText.getText();
411         if (KsCardInterfaceUtil.ksRequestCard(result)) {
412             String cardPhyId = result.getProperty("cardPhyId");
413             boolean success = false;
414             if (KsCardInterfaceUtil.ksPayCancelProcess(refNo, mac, cardPhyId, cardno, cardBal, dpsCnt)) {
415                 if (KsCardInterfaceUtil.ksCardBalanceDec(result)) {
416                     if (KsCardInterfaceUtil.ksPayProcess()) {
417                         success = true;
418                     }
419                 }
420             }
421             KsCardInterfaceUtil.getFieldValue("CARDBAL", result);
422             if (success) {
423                 balanceText.setText(String.valueOf(Double.parseDouble(result.getProperty("CARDBAL")) / 100));
424                 reverseBtn.setEnabled(false);
425                 Payrecord payrecord = getPayrecord();
426                 //1 扣款，0 冲正
427                 payrecord.setTransType("0");
428                 if (PayrecordService.savePayrecord(payrecord)) {
429                     KsCardInterfaceUtil.ksBeep();
430                 } else {
431                     JOptionPane.showMessageDialog(this, "保存付费记录失败", "Error", JOptionPane.ERROR_MESSAGE);
432                 }
433             } else {
434                 JOptionPane.showMessageDialog(this, "冲正失败", "Error", JOptionPane.ERROR_MESSAGE);
435             }
436         } else {
437             JOptionPane.showMessageDialog(this, "寻卡失败", "Error", JOptionPane.ERROR_MESSAGE);
438         }
