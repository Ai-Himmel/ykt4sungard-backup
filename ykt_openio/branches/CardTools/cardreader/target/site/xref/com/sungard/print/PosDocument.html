37       *
38       * @param graphics2D .
39       * @param pageFormat .
40       * @param pageIndex  .
41       */
42      @Override
43      public void drawCurrentPage(Graphics2D graphics2D, PageFormat pageFormat, int pageIndex) {
44          String content = this.getPageContent(pageIndex);
45          if (StringUtils.isBlank(content)) {
46              return;
47          }
48          String[] data = StringUtils.split(content, "<br>");
49          int fontSize = graphics2D.getFont().getSize();
50          //startRow=1代表从第一行开始打印，startColumn=0代表从第一列开始打印
51          int startRow = Configuration.getIntProperty(Constant.START_ROW, 1) * fontSize;
52          int startColumn = (Configuration.getIntProperty(Constant.START_COLUMN, 1) - 1) * fontSize;
53          for (String line : data) {
54              graphics2D.drawString(line, startColumn, startRow);
55              startRow += graphics2D.getFont().getSize();
56          }
57      }
