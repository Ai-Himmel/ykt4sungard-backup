46      private void button6ActionPerformed(ActionEvent e) {
47          JButton source = (JButton) e.getSource();
48          if (openPort) {
49              if (KsCardInterfaceUtil.ksClosePort()) {
50                  source.setText("打开端口");
51                  openPort = false;
52              } else {
53                  JOptionPane.showMessageDialog(CardReaderForm.this, "关闭端口失败", "Error", JOptionPane.ERROR_MESSAGE);
54              }
55          } else {
56              String port = comboBox1.getSelectedItem().toString();
57              String baud = comboBox2.getSelectedItem().toString();
58              if (StringUtils.isEmpty(port) || StringUtils.isEmpty(baud)) {
59                  JOptionPane.showMessageDialog(CardReaderForm.this, "请选择端口和波特率", "Warning", JOptionPane.WARNING_MESSAGE);
60              } else {
61                  if (KsCardInterfaceUtil.ksOpenPort(Integer.parseInt(port), Integer.parseInt(baud))) {
62                      source.setText("关闭端口");
63                      openPort = true;
64                  } else {
65                      JOptionPane.showMessageDialog(CardReaderForm.this, "打开端口失败", "Error", JOptionPane.ERROR_MESSAGE);
66                  }
67              }
68          }
69      }

140     private void button3ActionPerformed(ActionEvent e) {
141         String tramsAmt = textField4.getText();
142         if (StringUtils.isBlank(tramsAmt)) {
143             JOptionPane.showMessageDialog(CardReaderForm.this, "请输入扣款金额", "Warning", JOptionPane.WARNING_MESSAGE);
144         } else {
145             KsCardInterfaceUtil.ksRequestCard(result);
146             String cardPhyId = result.getProperty("cardPhyId");
147             int cardno = Integer.parseInt(result.getProperty("CARDNO"));
148             int cardBal = Integer.parseInt(result.getProperty("CARDBAL"));
149             int payCnt = Integer.parseInt(result.getProperty("PAYCNT"));
150             KsCardInterfaceUtil.ksPayPrepare(cardPhyId, cardno, cardBal, payCnt, Integer.parseInt(tramsAmt), result);
151             KsCardInterfaceUtil.ksCardBalanceDec(result);
152             KsCardInterfaceUtil.ksPayProcess();
153             String refNo = result.getProperty("refNo");
154             String mac = result.getProperty("mac");
155             textField5.setText(refNo);
156             textField6.setText(mac);
157 
158             KsCardInterfaceUtil.getFieldValue("CARDBAL", result);
159             textField3.setText(result.getProperty("CARDBAL"));
160 
161             KsCardInterfaceUtil.ksBeep();
162         }
163     }

165     private void button4ActionPerformed(ActionEvent e) {
166         int cardno = Integer.parseInt(result.getProperty("CARDNO"));
167         int cardBal = Integer.parseInt(result.getProperty("CARDBAL"));
168         KsCardInterfaceUtil.getFieldValue("DPSCNT", result);
169         int dpsCnt = Integer.parseInt(result.getProperty("DPSCNT"));
170         String refNo = textField5.getText();
171         String mac = textField6.getText();
172         KsCardInterfaceUtil.ksRequestCard(result);
173         String cardPhyId = result.getProperty("cardPhyId");
174         KsCardInterfaceUtil.ksPayCancelProcess(refNo, mac, cardPhyId, cardno, cardBal, dpsCnt);
175         KsCardInterfaceUtil.ksCardBalanceDec(result);
176         KsCardInterfaceUtil.ksPayProcess();
177         textField7.setText(String.valueOf(cardBal));
178         KsCardInterfaceUtil.getFieldValue("CARDBAL", result);
179         textField8.setText(result.getProperty("CARDBAL"));
180 
181         KsCardInterfaceUtil.ksBeep();
182     }
