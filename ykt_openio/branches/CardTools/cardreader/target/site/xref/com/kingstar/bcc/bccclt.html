336     static {
337         try {
338             String userDir = System.getProperty("user.dir");
339             String dll = userDir + File.separator + "bccclt.dll";
340             File dllFile = new File(dll);
341             if (!dllFile.exists()) {
342                 throw new RuntimeException("无法加载动态库：" + dll);
343             }
