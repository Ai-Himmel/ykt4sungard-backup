25      public static void exportExcel(File file, TableModel model) throws IOException {
26          if (file == null) {
27              return;
28          }
29          Workbook wb = new HSSFWorkbook();
30          Sheet sheet = wb.createSheet("sheet");
31  
32          int columnCount = model.getColumnCount();
33          int rowCount = model.getRowCount();
34          for (int i = 0; i <= rowCount; i++) {
35              Row row = sheet.createRow(i);
36              for (int j = 0; j < columnCount; j++) {
37                  Cell cell = row.createCell(j);
38                  cell.setCellType(Cell.CELL_TYPE_STRING);
39                  if (i == 0) {
40                      //create header
41                      cell.setCellValue(model.getColumnName(j));
42                  } else {
43                      Object value = model.getValueAt(i - 1, j);
44                      cell.setCellValue(value == null ? "" : value.toString());
45                  }
46              }
47          }
48          // Write the output to a file
49          FileOutputStream fileOut = new FileOutputStream(file);
50          wb.write(fileOut);
51          fileOut.close();
52      }

54      public static File saveAs(Component parent) {
55          File file = null;
56          JFileChooser fileChooser = new JFileChooser();
57          fileChooser.setAcceptAllFileFilterUsed(false);
58          // 添加excel文件的过滤器
59          fileChooser.addChoosableFileFilter(new ExportFileFilter("Excel 97-2003 工作簿(*.xls)", ".xls"));
60          // 下面这句是去掉显示所有文件这个过滤器。
61          int result = fileChooser.showSaveDialog(parent);
62          // JFileChooser.APPROVE_OPTION是个整型常量，代表0。就是说当返回0的值我们才执行相关操作，否则什么也不做。
63          if (result == JFileChooser.APPROVE_OPTION) {
64              file = fileChooser.getSelectedFile();
65              if (file.exists()) {
66                  if (!file.renameTo(file)) {
67                      file = null;
68                      JOptionPane.showMessageDialog(parent, "保存失败，文件被占用", "Error", JOptionPane.ERROR_MESSAGE);
69                  }
70              } else {
71                  String fileName = file.getAbsolutePath();
72                  if (!fileName.endsWith(".xls")) {
73                      fileName += ".xls";
74                  }
75                  file = new File(fileName);
76              }
77          }
78          return file;
79      }
