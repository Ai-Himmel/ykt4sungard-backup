101     private void portAction(ActionEvent e) {
102         JButton source = (JButton) e.getSource();
103         if (openPort) {
104             if (KsCardInterfaceUtil.ksClosePort()) {
105                 source.setText("打开端口");
106                 openPort = false;
107             } else {
108                 JOptionPane.showMessageDialog(this, "关闭端口失败", "Error", JOptionPane.ERROR_MESSAGE);
109             }
110         } else {
111             String port = portBox.getSelectedItem().toString();
112             String baud = baudBox.getSelectedItem().toString();
113             if (StringUtils.isEmpty(port) || StringUtils.isEmpty(baud)) {
114                 JOptionPane.showMessageDialog(this, "请选择端口和波特率", "Warning", JOptionPane.WARNING_MESSAGE);
115             } else {
116                 if (KsCardInterfaceUtil.ksOpenPort(Integer.parseInt(port), Integer.parseInt(baud))) {
117                     source.setText("关闭端口");
118                     openPort = true;
119                 } else {
120                     JOptionPane.showMessageDialog(this, "打开端口失败", "Error", JOptionPane.ERROR_MESSAGE);
121                 }
122             }
123         }
124     }

126     private void signAction(ActionEvent e) {
127         if (KsCardInterfaceUtil.ksSignIn(result)) {
128             String shopId = result.getProperty("shopId");
129             String shopName = result.getProperty("shopName");
130             shopIdText.setText(shopId);
131             shopNameText.setText(shopName);
132             JOptionPane.showMessageDialog(this, "读卡器签到成功[商户ID=" + shopId + ",商户名称=" + shopName + "]", "Info", JOptionPane.INFORMATION_MESSAGE);
133         } else {
134             shopIdText.setText("");
135             shopNameText.setText("");
136             String errMsg = result.getProperty("errMsg");
137             JOptionPane.showMessageDialog(this, "读卡器签到失败[" + errMsg + "]", "Error", JOptionPane.ERROR_MESSAGE);
138         }
