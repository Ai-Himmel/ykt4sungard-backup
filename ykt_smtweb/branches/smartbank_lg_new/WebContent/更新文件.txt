增加银行卡检查功能
CREATE TABLE DB2INST1.t_bank_check
 (id           INTEGER         NOT NULL  GENERATED ALWAYS
    AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20, NO CYCLE, NO ORDER),
  card_head    VARCHAR(20)     NOT NULL,
  card_length  INTEGER,
  bank_name    VARCHAR(100),
  bank_type    VARCHAR(100)
 )
  DATA CAPTURE NONE
  IN USERSPACE1;

#SYNC 10;

ALTER TABLE DB2INST1.t_bank_check
  LOCKSIZE ROW
  APPEND OFF
  NOT VOLATILE;

#SYNC 20;

COMMENT ON DB2INST1.t_bank_check
 (id IS '主键自增长',
  card_head IS '卡头',
  card_length IS '卡的长度',
  bank_name IS '银行名称',
  bank_type IS '银行类别'
 );

#SYNC 30;

ALTER TABLE DB2INST1.t_bank_check
  ADD CONSTRAINT pk_bank_check PRIMARY KEY
   (id
   );

#SYNC 40;

COMMENT ON CONSTRAINT DB2INST1.t_bank_check.pk_bank_check
  IS '住键约束';

#SYNC 50;

RUNSTATS ON TABLE DB2INST1.t_bank_check
  AND INDEXES ALL 
  SHRLEVEL REFERENCE;

#SYNC 60;
