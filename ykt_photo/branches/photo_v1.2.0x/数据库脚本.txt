CREATE TABLE YKT_CUR.T_CIF_DEPT
 (DEPT_CODE        VARCHAR(10)     NOT NULL,
  IS_ACADEMY       CHARACTER(1),
  DEPT_NAME        VARCHAR(100),
  E_NAME           VARCHAR(100),
  PARENTDEPT_CODE  VARCHAR(10),
  DEPT_LEVEL       INTEGER,
  IS_AVAILABILITY  CHARACTER(1),
  COMMENTS         VARCHAR(100)
 )
  DATA CAPTURE NONE
  IN USERSPACE1;

ALTER TABLE YKT_CUR.T_CIF_DEPT
  LOCKSIZE ROW
  APPEND OFF
  NOT VOLATILE;

COMMENT ON TABLE YKT_CUR.T_CIF_DEPT
  IS '单位信息';

COMMENT ON YKT_CUR.T_CIF_DEPT
 (DEPT_CODE IS '单位代码',
  IS_ACADEMY IS '是否院系',
  DEPT_NAME IS '单位名称',
  E_NAME IS '英文名称',
  PARENTDEPT_CODE IS '上级单位',
  DEPT_LEVEL IS '单位级别',
  IS_AVAILABILITY IS '是否有效'
 );

ALTER TABLE YKT_CUR.T_CIF_DEPT
  ADD CONSTRAINT P_KEY_1 PRIMARY KEY
   (DEPT_CODE
   );
